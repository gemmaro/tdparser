def initialize(args = nil, &block)
  enumerator = Enumerator.new do |y|
    if args
      args.each { |arg| y << arg }
    else
      block.call(y)
    end
  end
  @enumerator = enumerator

  @buffer = []
end

def shift
  if @buffer.empty?
    (self.next if next?)
  else
    @buffer.shift
  end
end
